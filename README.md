## Installation

```bash
npm install @alexandermann/ui-toolkit
```

or

```bash
yarn add @alexandermann/ui-toolkit
```

## Usage

```jsx
import { Button, Card } from '@alexandermann/ui-toolkit'

function App() {
  return (
    <div>
      <Button variant="primary">Click me</Button>
      <Card>
        <h2>Card Title</h2>
        <p>Card content goes here</p>
      </Card>
    </div>
  )
}
```
